Necessary JavaScript Code For Making Image Slider

					script type="text/javascript">
						var imagecount = 1;
						var total = 4;
						function slider(x){
							var image = document.getElementById("img");
							imagecount = imagecount + x;
							image.src = "pics/aftab"+ imagecount +".jpg";
							if(imagecount > total){
								imagecount = 1;
							}
							if(imagecount < 1){
								imagecount = total;
							}
						}
						window.setInterval(function sliderA(){
							var image = document.getElementById("img");
							imagecount = imagecount + 1;
							image.src = "pics/aftab"+ imagecount +".jpg";
							if(imagecount > total){
								imagecount = 1;
							}
							if(imagecount < 1){
								imagecount = total;
							}
						}, 5000);
					
					script>
